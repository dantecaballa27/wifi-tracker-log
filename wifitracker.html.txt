        <table>
            <thead>
                <tr>
                    <th>User Name</th>
                    <th>Time In</th>

                    <th>Status</th>
                </tr>

                </tbody>

    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const userNameInput = document.getElementById('userNameInput');
            const logInBtn = document.getElementById('logInBtn');
            const logOutBtn = document.getElementById('logOutBtn');
            const userLogBody = document.getElementById('userLogBody');

            // Load data from browser's localStorage or start with an empty array
            let userLog = JSON.parse(localStorage.getItem('wifiUserLog')) || [];

            // Function to save the log data
            function saveLog() {
                localStorage.setItem('wifiUserLog', JSON.stringify(userLog));
            }

            // Function to format the current time
            function formatTime(date) {
                return date.toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit', second: '2-digit' });
            }

            // Function to render the table rows
            function renderLog() {
                userLogBody.innerHTML = ''; // Clear existing rows
                userLog.forEach(log => {
                    const row = userLogBody.insertRow();

                    row.insertCell().textContent = log.name;
                    row.insertCell().textContent = log.timeIn;
                    row.insertCell().textContent = log.timeOut || 'â€”'; // Display dash if not logged out
                    
                    const statusCell = row.insertCell();
                    if (log.timeOut) {
                        statusCell.textContent = 'Logged Out';
                        statusCell.style.color = '#dc3545';
                    } else {
                        statusCell.textContent = 'Active';
                        statusCell.style.color = '#28a745';
                    }
                });
            }

            // Log In handler
            logInBtn.addEventListener('click', () => {
                const userName = userNameInput.value.trim();
                if (!userName) {
                    alert('Please enter a user name to log in.');
                    return;
                }

                // Check if user is already logged in (no timeOut)
                const existingEntry = userLog.find(log => log.name === userName && !log.timeOut);
                if (existingEntry) {
                    alert(${userName} is already logged in.);
                    return;
                }

                const newEntry = {
                    name: userName,
                    timeIn: formatTime(new Date()),
                    timeOut: null // Null indicates active
                };
                userLog.push(newEntry);
                saveLog();
                renderLog();
                userNameInput.value = '';
            });

            // Log Out handler
            logOutBtn.addEventListener('click', () => {
                const userName = userNameInput.value.trim();
                if (!userName) {
                    alert('Please enter a user name to log out.');
                    return;
                }

                // Find the latest active log entry for this user
                const activeEntry = userLog
                    .slice() // Create a shallow copy
                    .reverse() // Search from the most recent entry
                    .find(log => log.name === userName && !log.timeOut);

                if (!activeEntry) {
                    alert(${userName} is not currently logged in or all sessions are closed.);
                    return;
                }

                // Update the timeOut for the found entry
                activeEntry.timeOut = formatTime(new Date());

                saveLog();
                renderLog();
                userNameInput.value = '';
            });

            // Initial render on load
            renderLog();
        });
    </script>
